#include <PN532_HSU.h>
#include <PN532.h>

// PN532 connected to Serial1 on Mega (RX1=19, TX1=18)
PN532_HSU pn532hsu(Serial1);
PN532 nfc(pn532hsu);

String tagId = "None", dispTag = "None";
byte nuidPICC[4];

void setup(void) {
  Serial.begin(115200);     // For debugging via USB
  Serial1.begin(115200);    // PN532 default HSU baud rate
  Serial.println("Hello Maker! (Mega + PN532 via Serial1)");

  nfc.begin();

  uint32_t versiondata = nfc.getFirmwareVersion();
  if (!versiondata) {
    Serial.println("❌ Didn't Find PN53x Module");
    while (1); // Halt
  }

  Serial.print("✅ Found chip PN5"); Serial.println((versiondata >> 24) & 0xFF, HEX);
  Serial.print("Firmware ver. "); Serial.print((versiondata >> 16) & 0xFF, DEC);
  Serial.print('.'); Serial.println((versiondata >> 8) & 0xFF, DEC);

  nfc.SAMConfig(); // Configure to read RFID tags
  Serial.println("Waiting for an ISO14443A Card ...");
}

void loop() {
  readNFC();
}

void readNFC() {
  uint8_t uid[7] = { 0 };   // Buffer for UID
  uint8_t uidLength = 0;    // UID length

  bool success = nfc.readPassiveTargetID(PN532_MIFARE_ISO14443A, uid, &uidLength);

  if (success) {
    Serial.print("UID Length: "); Serial.print(uidLength); Serial.println(" bytes");
    Serial.print("UID Value: ");
    for (uint8_t i = 0; i < uidLength; i++) {
      nuidPICC[i] = uid[i];
      Serial.print(" "); Serial.print(uid[i], DEC);
    }
    Serial.println();

    tagId = tagToString(nuidPICC);
    dispTag = tagId;
    Serial.print(F("tagId is : "));
    Serial.println(tagId);
    Serial.println();

    delay(1000); // wait 1 second
  } else {
    Serial.println("⏳ Timed out! Waiting for a card...");
    delay(300);
  }
}

String tagToString(byte id[4]) {
  String tagId = "";
  for (byte i = 0; i < 4; i++) {
    tagId += String(id[i]);
    if (i < 3) tagId += ".";
  }
  return tagId;
}
